
Table Parkhouse {
  id integer [primary key]
  name varchar
  capacity integer
  address varchar
  rating int
  opening datetime
  closing datetime
  car_height int
  max_stay_time integer
}

Table users {
  id integer [primary key]
  name varchar
  car_id integer
  email varchar
  phone_num varchar
  role varchar
  username varchar
  password varchar
}

Table reservations {
  id integer [primary key]
  start_time datetime
  active datetime
  inactive datetime
  reservation_time_hour datetime
  sum int
  reservation_owner_id integer
  park_slot integer
  park_house_id integer
}

Table car{
  car_id integer [primary key]
  plate varchar
  height int
  type varchar
  owner_id integer
}

Ref: users.car_id > car.car_id
Ref: users.id > reservations.reservation_owner_id
Ref: users.id > car.owner_id
Ref: reservations.park_house_id > Parkhouse.id
